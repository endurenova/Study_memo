/ : 나눴을 때, 몫만 출력

% : 나눴을 때, 나머지 값만 출력

SELECT Country || " " || City || " " || Address AS 주소 FROM Customers;

연결 연산자

|| : Oracle

+ : mSQL

" " : mySQL

ex)

SELECT ProductName AS 제품이름, Price AS 기존가, Price*0.2 AS 할인된가격, Price*0.8 AS 최종가 FROM Products

& (and) | (or)

True and False = False

True or False = True

not True = False

SELECT 5 & 7;

→ 5 (2진법 101)

SELECT 5 | 7;

→ 7 (2진법 111)

SELECT * FROM Customers WHERE CustomerID BETWEEN 30 AND 50;

SELECT * FROM Customers WHERE CustomerID IN (10, 20, 30);

SELECT * FROM Customers WHERE CustomerID LIKE '1';

_ : wild card (개수 만큼 체크됨)

SELECT * FROM Customers WHERE CustomerID LIKE '_1';

- CustomerID 가 x1인 리스트

SELECT * FROM Customers WHERE CustomerID LIKE '1_';

- CustomerID 가 1x인 리스트

SELECT * FROM Customers WHERE CustomerID LIKE '%1';

- CustomerID에서 1이 뒤에 들어가는 리스트

SELECT * FROM Customers WHERE CustomerID LIKE '1%';

- CustomerID에서 1이 앞에 들어가는 리스트

SELECT * FROM Customers WHERE CustomerID IS NULL;
